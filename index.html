📋 คลิกเพื่อดู Apps Script Code (copy ทั้งหมด) 
function doGet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName("Data");

// Helper: safely get numeric value
function n(row, col) {
var v = sh.getRange(row, col).getValue();
return (typeof v === “number” && !isNaN(v)) ? v : 0;
}

var data = {
SET:   { open:n(2,2), high:n(3,2), low:n(4,2), close:n(5,2), prevClose:n(6,2), vol:n(7,2) },
SET50: { open:n(2,4), high:n(3,4), low:n(4,4), close:n(5,4), prevClose:n(6,4), vol:n(7,4) },
N225:  { open:n(2,6), high:n(3,6), low:n(4,6), close:n(5,6), prevClose:n(6,6), vol:n(7,6) },
HSI:   { open:n(2,8), high:n(3,8), low:n(4,8), close:n(5,8), prevClose:n(6,8), vol:n(7,8) },
updated: new Date().toISOString()
};

return ContentService
.createTextOutput(JSON.stringify(data))
.setMimeType(ContentService.MimeType.JSON);
}
